```bash
#
# Usage:
#   total=$(sum_numbers 3 7 12)   # total == 22
#   total=$(sum_numbers "$@")     # sum all script arguments
# ------------------------------------------------------------
sum_numbers() {
    local sum=0

    # Iterate over every argument
    for n in "$@"; do
        # Verify that the argument looks like an integer or a float.
        # If it doesn't, we skip it (you could also choose to abort).
        if [[ $n =~ ^-?[0-9]+([.][0-9]+)?$ ]]; then
            # Bash can only do integer arithmetic with (( )), so for
            # full floating‚Äëpoint support we rely on bc.
            if [[ $n == *.* ]]; then
                # Use bc for decimal numbers

            else
                (( sum += n ))
            fi
        else
            echo "Warning: '$n' is not a number ‚Äì ignored" >&2

    done

    printf "%s\n" "$sum"
}
```

3. **Add the value**

   * For **floating‚Äëpoint numbers**, we delegate the addition to `bc -l`, which handles arbitrary‚Äëprecision decimal math.
4. **Output** ‚Äì the final sum is printed with `printf`, which makes it easy to capture via command substitution: `total=$(sum_numbers ‚Ä¶)`.
